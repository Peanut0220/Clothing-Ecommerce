CREATE TABLE CUSTOMER (CUSTID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, CUSTUSERNAME VARCHAR(50) NOT NULL, CUSTFIRSTNAME VARCHAR(50) NOT NULL, CUSTLASTNAME VARCHAR(50) NOT NULL, CUSTEMAIL VARCHAR(50) NOT NULL, CUSTPASSWORD VARCHAR(50) NOT NULL, CUSTPHONENUM VARCHAR(12), PRIMARY KEY (CUSTID));
CREATE TABLE STAFF (STAFFID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, STAFFUSERNAME VARCHAR(50) NOT NULL, STAFFFIRSTNAME VARCHAR(50) NOT NULL, STAFFLASTNAME VARCHAR(50) NOT NULL, STAFFEMAIL VARCHAR(50) NOT NULL, STAFFPASSWORD VARCHAR(50) NOT NULL, STAFFPHONENO VARCHAR(12), STAFFCREATEDDATE DATE, PRIMARY KEY (STAFFID));
CREATE TABLE PRODUCT (PRODID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, PRODNAME VARCHAR(50) NOT NULL, PRODPRICE DOUBLE NOT NULL, PRODDESC VARCHAR(200) NOT NULL, CATEGORYNAME VARCHAR(60), PRODIMAGE BLOB, QUANTITY INTEGER, PRIMARY KEY (PRODID));
CREATE TABLE ORDERS (ORDERID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, ORDERDATE DATE, ORDERADDRESS VARCHAR(150) NOT NULL, ORDERSTATUS VARCHAR(50) NOT NULL, SHIPPINGFEES DOUBLE NOT NULL, SUBTOTAL DOUBLE NOT NULL, TOTAL DOUBLE NOT NULL, CUSTID INTEGER NOT NULL, ESTIMDATE DATE, TRACKINGNUM INTEGER, PRIMARY KEY (ORDERID));
CREATE TABLE ORDERDETAILS (ORDERDETAILSID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, QUANTITY INTEGER NOT NULL, ORDERID INTEGER NOT NULL, PRODID INTEGER NOT NULL, PRIMARY KEY (ORDERDETAILSID));
CREATE TABLE CART (CARTID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, CUSID INTEGER, PRIMARY KEY (CARTID));
CREATE TABLE CARTDETAILS (CARTDETAILSID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, CARTID INTEGER NOT NULL, PRODID INTEGER NOT NULL, QUANTITY INTEGER NOT NULL, PRIMARY KEY (CARTDETAILSID));
CREATE TABLE PRODUCTTOTAL (PRODTOTALID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, PRODTOTAL INTEGER, PRODID INTEGER, PRIMARY KEY (PRODTOTALID));
